def is_prime(num):
    if num <= 1:
        return False
    for i in range(2, int(num ** 0.5) + 1):
        if num % i == 0:
            return False
    return True

print("Prime numbers less than 20:")
for i in range(2, 20):
    if is_prime(i):
        print(i, end=" ")

12. Write a python program to find factorial of a numbers using recursion. 

def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

num = int(input("Enter a number: "))

if num < 0:
    print("Factorial is not defined for negative numbers.")
elif num == 0:
    print("The factorial of 0 is 1.")
else:
    result = factorial(num)
    print(f"The factorial of {num} is {result}.")

def fibonacci_sequence(n):
    fib_sequence = [0, 1]
    while len(fib_sequence) < n:
        fib_sequence.append(fib_sequence[-1] + fib_sequence[-2])
    return fib_sequence

def fibonacci_number(n):
    if n <= 0:
        return "Invalid input. Please enter a positive integer."

17. Write a program that inputs a text file. The program should print all the unique words in the file in alphabetical order 

def print_unique_words(file_name):
    unique_words = set()
    try:
        with open(file_name, "r") as file:
            for line in file:
                words = line.split()
                for word in words:
                    unique_words.add(word.strip().lower())
        unique_words_list = sorted(list(unique_words))
        print("Unique words in alphabetical order:")
        for word in unique_words_list:
            print(word)
    except FileNotFoundError:
        print("File not found. Please enter a valid file name.")
    except Exception as e:
        print("An error occurred:", str(e))


file_name = input("Enter the name of the text file: ")

18. write a python class to convert an integer to a roman numeral 


class IntegerToRoman:
  def __init__(self):
      self.roman_numerals = {
          1: 'I', 4: 'IV', 5: 'V', 9: 'IX', 10: 'X', 40: 'XL',
          50: 'L', 90: 'XC', 100: 'C', 400: 'CD', 500: 'D', 900: 'CM', 1000: 'M'
      }
  def to_roman(self, num):
      if num <= 0:
          return "Invalid input. Please enter a positive integer."
      roman_numeral = ''
      for value, numeral in sorted(self.roman_numerals.items(), reverse=True):
          while num >= value:
              roman_numeral += numeral
              num -= value
      return roman_numeral
      
      
converter = IntegerToRoman()
num = int(input("Enter an integer to convert to Roman numeral: "))
roman_numeral = converter.to_roman(num)
print(f"The Roman numeral representation of {num} is: {roman_numeral}")
